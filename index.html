                showModal('error', 'Por favor, complete los campos obligatorios (Número de Expediente, Fecha de Inicio, Tipo de Caso, Estado del Expediente).');

                return;

            }

 

            const existingIndex = expedientes.findIndex(exp => exp.id === newExpediente.id);

            if (existingIndex > -1) {

                // Update existing

                expedientes[existingIndex] = newExpediente;

                showModal('success', 'Expediente actualizado con éxito.');

            } else {

                // Check for duplicate numeroExpediente only for new entries

                if (expedientes.some(exp => exp.numeroExpediente === newExpediente.numeroExpediente)) {

                    showModal('error', `El expediente "${newExpediente.numeroExpediente}" ya existe.`);

                    return;

                }

                // Add new

                expedientes.push(newExpediente);

                showModal('success', 'Expediente creado con éxito.');

            }

            saveExpedientes();

            hideExpedienteForm();

            showView('dashboard');

        }

                return matchesSearchTerm && matchesTipoCaso && matchesAreaDerecho && matchesJuzgado;

            }).sort((a, b) => a.numeroExpediente.localeCompare(b.numeroExpediente));

 

            if (filteredExpedientes.length === 0) {

        function calcularSaldoPendiente(exp) {

            const honorarios = parseFloat(exp.honorariosPactados || 0);

            const gastos = parseFloat(exp.gastosProceso || 0);

            const totalPagos = (exp.pagosRecibidos || []).reduce((sum, pago) => sum + parseFloat(pago.monto || 0), 0);

            return (honorarios + gastos - totalPagos);

        }

        function showModal(type, message, onConfirm = null) {

            const modalContent = document.getElementById('modalContent');

            const modalIconContainer = document.getElementById('modalIconContainer');

            const modalMessageEl = document.getElementById('modalMessage');

            const modalButtons = document.getElementById('modalButtons');

 

            modalMessageEl.textContent = message;

 

            // Reset styles and icons

            modalContent.className = 'bg-white p-6 rounded-xl shadow-2xl max-w-sm w-full text-center transform scale-100 animate-zoomIn border-t-4';

            modalIconContainer.innerHTML = '';

 

            switch (type) {

                case 'success':

                    modalContent.classList.add('border-green-500');

                    modalIconContainer.innerHTML = '<i data-lucide="check-circle" class="w-12 h-12 text-green-500"></i>';

                    break;

                case 'error':

                    modalContent.classList.add('border-red-500');

                    modalIconContainer.innerHTML = '<i data-lucide="x-circle" class="w-12 h-12 text-red-500"></i>';

                    break;

                case 'confirm':

                    modalContent.classList.add('border-yellow-500');

                    modalIconContainer.innerHTML = '<i data-lucide="help-circle" class="w-12 h-12 text-yellow-500"></i>';

                    break;

                default:

                    modalContent.classList.add('border-blue-500');

                    modalIconContainer.innerHTML = '<i data-lucide="info" class="w-12 h-12 text-blue-500"></i>';

            }

 

            modalButtons.innerHTML = '';

            if (type === 'confirm' && onConfirm) {

                const confirmButton = document.createElement('button');

                confirmButton.className = 'bg-blue-500 hover:bg-blue-600 text-white font-bold py-2 px-4 rounded-lg';

                confirmButton.textContent = 'Confirmar';

                confirmButton.onclick = () => {

                    onConfirm();

                    hideConfirmationModal();

                };

 

                const cancelButton = document.createElement('button');

                cancelButton.className = 'bg-gray-300 hover:bg-gray-400 text-gray-800 font-bold py-2 px-4 rounded-lg';

                cancelButton.textContent = 'Cancelar';

                cancelButton.onclick = hideConfirmationModal;

 

                modalButtons.appendChild(confirmButton);

                modalButtons.appendChild(cancelButton);

            } else {

                const closeButton = document.createElement('button');

                closeButton.className = 'bg-blue-500 hover:bg-blue-600 text-white font-bold py-2 px-4 rounded-lg';

                closeButton.textContent = 'Cerrar';

                closeButton.onclick = hideConfirmationModal;

                modalButtons.appendChild(closeButton);

            }

 

            document.getElementById('confirmationModal').classList.remove('hidden');

            lucide.createIcons();

        }

        async function downloadHonorarios(format, expedienteId) {

            const exp = expedientes.find(e => e.id === expedienteId);

            if (!exp) {

                showModal('error', 'Expediente no encontrado.');

                return;

            }

 

            const representados = [...(exp.actores || []), ...(exp.demandados || [])]

                .filter(p => p.esRepresentado)

                .map(p => p.nombre)

                .join(', ');

           

            const totalPagado = (exp.pagosRecibidos || []).reduce((sum, pago) => sum + parseFloat(pago.monto || 0), 0);

            const saldo = calcularSaldoPendiente(exp);

 

            const data = {

                numeroExpediente: exp.numeroExpediente,

                tipoCaso: exp.tipoCaso,

                fechaInicio: exp.fechaInicio,

                estadoExpediente: exp.estadoExpediente,

                representados: representados || 'N/A',

                honorariosPactados: parseFloat(exp.honorariosPactados || 0).toFixed(2),

                gastosProceso: parseFloat(exp.gastosProceso || 0).toFixed(2),

                totalPagado: totalPagado.toFixed(2),

                saldoPendiente: saldo.toFixed(2),

                pagosRecibidos: exp.pagosRecibidos.map(p => ({ fecha: p.fecha, monto: parseFloat(p.monto || 0).toFixed(2), concepto: p.concepto || 'N/A' })),

                notas: exp.notas.map(n => ({ fecha: n.fecha, contenido: n.contenido }))

            };

 

            if (format === 'pdf') {

                generatePdfReport(data);

            } else if (format === 'word') {

                generateWordReport(data);

            }

        }
